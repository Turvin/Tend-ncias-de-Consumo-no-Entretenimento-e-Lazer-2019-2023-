Crescimento Pós-Pandemia % = 
VAR PrePandemia = CALCULATE(
    [Receita Total],
    entretenimento_consumo_brasil[period] = "Pré-Pandemia"
)
VAR PosPandemia = CALCULATE(
    [Receita Total],
    entretenimento_consumo_brasil[period] = "Pós-Pandemia"
)
RETURN DIVIDE(PosPandemia - PrePandemia, PrePandemia)
